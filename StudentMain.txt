{

                    s.setAge(sc.nextInt());
                    ss.addStudent(s);
                    break;
                case 2:
                    ss.displayStudents();
                    break;
                case 3:
                    StudentModel us = new StudentModel();
                    System.out.print("Enter Student Name To Update : ");
                    us.setName(sc.next());
                    System.out.print("Enter New Age : ");
                    us.setAge(sc.nextInt());
                    ss.updateStudent(us);
                    break;
                case 4:
                    System.out.print("Enter Student Name To Delete : ");
                    String name = sc.next();
                    ss.deleteStudent(name);
                    break;
                case 5:
                    System.out.println("Program Exited Successfully");
                    break;
                default:
                    System.out.println("Invalid Choice");
            }
        } while (choice != 5);

}


package student;
    public class StudentModel 
    {
        private String name;
        private int age;
        public StudentModel() 
        {

        }
        public StudentModel(String name, int age) 
        {
            this.name = name;
            this.age = age;
        }
        public String getName() 
        {
            return name;
        }
        public void setName(String name) 
        {
            this.name = name;
        }
        public int getAge() 
        {
            return age;
        }
        public void setAge(int age) 
        {
            this.age = age;
        }
    }
	
	
	
	package student;
import java.sql.*;
import java.util.ArrayList;
public class StudentDAO 
{
    public Connection dbconnection() throws ClassNotFoundException, SQLException 
    {
        Class.forName("org.postgresql.Driver");
        Connection c = DriverManager.getConnection("jdbc:postgresql://localhost:5432/Record","postgres","saravana1618");
        return c;
    }
    public void insert(StudentModel s) throws ClassNotFoundException, SQLException 
    {
        Connection c = dbconnection();
        PreparedStatement ps = c.prepareStatement("INSERT INTO Student(Name,Age) VALUES(?,?)");
        ps.setString(1, s.getName());
        ps.setInt(2, s.getAge());
        ps.executeUpdate();
    }
    public ArrayList<StudentModel> viewAll() throws ClassNotFoundException, SQLException 
    {
        ArrayList<StudentModel> al = new ArrayList<>();
        Connection c = dbconnection();
        PreparedStatement ps = c.prepareStatement("SELECT * FROM Student");
        ResultSet rs = ps.executeQuery();
         while (rs.next()) 
         {
            StudentModel s = new StudentModel();
            s.setName(rs.getString("Name"));
            s.setAge(rs.getInt("Age"));
            al.add(s);
        }
        return al;
    }
    public void update(StudentModel s) throws ClassNotFoundException, SQLException 
    {
        Connection c = dbconnection();
        PreparedStatement ps = c.prepareStatement("UPDATE Student SET Age=? WHERE Name=?");
        ps.setInt(1, s.getAge());
        ps.setString(2, s.getName());
        ps.executeUpdate();
    }
    public void delete(String name) throws ClassNotFoundException, SQLException 
    {
        Connection c = dbconnection();
        PreparedStatement ps = c.prepareStatement("DELETE FROM Student WHERE Name=?");
        ps.setString(1, name);
        ps.executeUpdate();
    }
}


package student;
import java.sql.SQLException;
import java.util.ArrayList;
public class StudentService 
{
    StudentDAO d = new StudentDAO();
    public void addStudent(StudentModel s) throws ClassNotFoundException, SQLException 
    {
        d.insert(s);
        System.out.println("Inserted");
    }
    public void displayStudents() throws ClassNotFoundException, SQLException 
    {
        ArrayList<StudentModel> al = d.viewAll();
        System.out.println("\nNAME     AGE");
        for (StudentModel s : al) 
        {
            System.out.println(s.getName() + "     " + s.getAge());
        }
    }
    public void updateStudent(StudentModel s) throws ClassNotFoundException, SQLException 
    {
        d.update(s);
        System.out.println("Updated");
    }
    public void deleteStudent(String name) throws ClassNotFoundException, SQLException 
    {
        d.delete(name);
        System.out.println("Deleted");
    }
}
